sub somesub {
    say 'Some real code...';
    eval {
        return 'Eval return is not highlighted.' if $a;
        say 'Some real code...';
        die 'And we are dead' if $b;
        say 'Some real code...';
        <READ>exit -1</READ> if $b;
        say 'Some real code...';
        42;
    };
    sub {
        return 'Anon sub return is not highlighted.' if $a;
        say 'Some real code...';
        die 'And we are dead' if $b;
        say 'Some real code...';
        <READ>exit -1</READ> if $b;
        say 'Some real code...';
        42;
    };
    do {
        return 'Do return is not highlighted.' if $a;
        say 'Some real code...';
        <READ>die 'And we are dead'</READ> if $b;
        say 'Some real code...';
        <READ>exit -1</READ> if $b;
        say 'Some real code...';
        42;
    };
    sort {
        return 'Sort return is not highlighted.' if $a;
        say 'Some real code...';
        die 'And we are dead' if $b;
        say 'Some real code...';
        <READ>exit -1</READ> if $b;
        say 'Some real code...';
        42;
    } @ARGV;

    grep {
        <READ>return 'grep should return'</READ> if $a;
        <READ>exit -1</READ> if $b;
    } @ARGV;

    map {
        <READ>return 'map should return'</READ> if $a;
        <READ>exit -1</READ> if $b;
    } @ARGV;

    "string should retrun ${
        <READ>return 'map should return'</READ> if $a;
        <READ>exit -1</READ> if $b;
    } text";

    <<EOM;
    some text
    ${
        <READ>return 'map should return'</READ> if $a;
        <READ>exit -1</READ> if $b;
        } here
EOM

    s/    some text
         ${
             <READ>return 'map should return'</READ> if $a;
             <READ>exit -1</READ> if $b;
             } here
    /
             <READ>return 'map should return'</READ> if $a;
             <READ>exit -1</READ> if $b;
/xe;

    <READ>ret<caret>urn 'this is a proper return'</READ> if $a;
    say 'Some real code...';
    <READ>die 'And we are dead'</READ> if $b;
    say 'Some real code...';
    <READ>exit 42</READ> if $b;
    say 'Some real code...';
    <READ>say 'This is a last statement exit point'</READ>;
}
